<script>
    function switchTab(tabName) {
        // Hilangkan kelas aktif dari semua tab button dan form content
        document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
        document.querySelectorAll('.form-content').forEach(form => form.classList.remove('active'));

        // Aktifkan tab dan form yang diklik
        if(tabName === 'stunting') {
            document.querySelector('button[onclick="switchTab(\'stunting\')"]').classList.add('active');
            document.getElementById('form-stunting').classList.add('active');
        } else {
            document.querySelector('button[onclick="switchTab(\'personality\')"]').classList.add('active');
            document.getElementById('form-personality').classList.add('active');
        }
    }

    // Trik Otomatis: Jika website mendeteksi hasil personality dikirim balik, buka tab personality secara otomatis
    {% if pers_text %}
        switchTab('personality');
    {% endif %}
